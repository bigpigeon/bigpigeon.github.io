i < b.N; i++ { TestLock() } } func BenchmarkTestDefer(b *testing.B) { for i := 0; i < b.N; i++ { TestDeferLock() } } func main() { test0() test1() test2() test3() test4() // 性能测试跑的时间太长无法在play中 // fmt.Println(testing.Benchmark(BenchmarkTest)) // fmt.Println(testing.Benchmark(BenchmarkTestDefer)) }
